function solution(s) {

  var countP = 0;
  var countY = 0;
  var searchP = "P";

  var Pcount = sUpper.indexOf(searchP);

    Pcount = sUpper.indexOf(searchP, Pcount + 1);

  while (Ycount !== -1) {
    countY++;
    Ycount = sUpper.indexOf(searchY, Ycount + 1);
  }

  if (countY == countP) {
    return true;
  } else return false;
}
